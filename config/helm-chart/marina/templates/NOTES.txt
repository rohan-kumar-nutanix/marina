  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "marina.fullname" . }})

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "marina.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
